95  template <typename T, typename Unused = void>
96  class SchemaCache
97  {
98  public:
99  static const SchemaDef& Get()
100  {
101  // The schema object can't be initialized as a global static
102  // because InitSchemaDef uses schema's and fields' metadata which
103  // are global static variables, and we can't depends on them being
104  // initialized before the schema. Instead we initialize the schema
105  // on the first call to Get().
106  // Note that older versions of GNU C++ don't handle rvalue argument
107  // forwarding in Boost call_once implementation so we are using
108  // the old trusty boost::bind.
109  call_once(flag, boost::bind(&AppendStructDef, &schema));
110  return schema;
111  }
112 
113  static void AppendStructDef(SchemaDef* s);
114 
115  private:
116  static SchemaDef schema;
117  static once_flag flag;
118  };

258  template <typename T>
259  uint16_t GetStructDef() const
260  {
261  size_t n;
262 
263  for (n = 0; n < _schema.structs.size(); ++n)
264  if (_schema.structs[n].metadata.qualified_name == schema<T>::type::metadata.qualified_name)
265  return static_cast<uint16_t>(n);
266 
267  detail::SchemaCache<T>::AppendStructDef(&_schema);
268 
269  BOOST_ASSERT(n == static_cast<uint16_t>(n));
270  return static_cast<uint16_t>(n);
271  }

278  StructDef& This() const

292 template <typename T, typename Unused>
293 void SchemaCache<T, Unused>::AppendStructDef(SchemaDef* s)
294 {
295  // To apply InitSchemaDef transform we need a reference to an
296  // object T. However the transform never accesses the object or
297  // its fields. We can't construct an actual object since we
298  // need to support stateful allocators that can't allocate from
299  // a default-constructed instance and containers which allocate in
300  // default constructor.
301  char o;
302  Apply(InitSchemaDef(*s), reinterpret_cast<const T&>(o));
303 }
