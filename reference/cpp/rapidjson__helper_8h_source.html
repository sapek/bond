26 // Adapter from Bond input stream to rapidjson read-only stream
27 template <typename Buffer>
28 class RapidJsonInputStream
29 {
30 public:
31  typedef char Ch;
32 
33  RapidJsonInputStream(typename boost::call_traits<Buffer>::reference input)
34  : input(&input),
35  count(0)
36  {
37  input.Read(current);
38  }
39 
40  RapidJsonInputStream(const RapidJsonInputStream& that, typename boost::call_traits<Buffer>::reference input)
41  : input(&input),
42  current(that.current),
43  count(that.count)
44  {}
45 
46  char Peek()
47  {
48  return current;
49  }
50 
51  size_t Tell() const
52  {
53  return count;
54  }
55 
56  char Take()
57  {
58  char c = current;
59 
60  if (!input->IsEof())
61  {
62  input->Read(current);
63  ++count;
64  }
65  else
66  {
67  current = '\x0';
68  }
69 
70  return c;
71  }
72 
73  // not implemented for read only stream
74  char* PutBegin() { BOOST_ASSERT(false); return 0; }
75  void Put(char) { BOOST_ASSERT(false); }
76  size_t PutEnd(char*) { BOOST_ASSERT(false); return 0; }
77 
78  RapidJsonInputStream& operator=(const RapidJsonInputStream& that)
79  {
80  // rapidjson reader makes a local copy of stream within some functions
81  // and assigns it back to its main stream variable before function exit.
82  BOOST_ASSERT(input == that.input);
83  current = that.current;
84  count = that.count;
85  return *this;
86  }
87 
88 private:
89  Buffer* input;
90  uint8_t current;
91  size_t count;
92 };

144 class JsonTypeMatching : boost::noncopyable
145 {
146 public:
147  JsonTypeMatching(BondDataType type, BondDataType schema, bool is_enum)
148  : matchesObject(type == BT_STRUCT && type == schema),
149  matchesArray((type == BT_MAP || type == BT_LIST || type == BT_SET) && type == schema),
150  matchesNull(type == BT_LIST && type == schema),
151  matchesInt(type >= BT_INT8 && type <= BT_INT64),
152  matchesInt64(type == BT_INT64),
153  matchesUint(type >= BT_UINT8 && type <= BT_UINT64),
154  matchesUint64(type == BT_UINT64),
155  matchesNumber(type >= BT_FLOAT && type <= BT_DOUBLE),
156  matchesString(type == BT_STRING || type == BT_WSTRING || is_enum),
157  matchesBool(type == BT_BOOL)
158  {}
159 
160 
161  bool TypeMatch(const rapidjson::Value& value) const
162  {
163  return ComplexTypeMatch(value) || BasicTypeMatch(value);
164  }
165 
166  bool ComplexTypeMatch(const rapidjson::Value& value) const
167  {
168  return ((value.IsObject() && matchesObject)
169  || (value.IsArray() && matchesArray)
170  || (value.IsNull() && matchesNull));
171  }
172 
173  bool BasicTypeMatch(const rapidjson::Value& value) const
174  {
175  return ((value.IsString() && matchesString)
176  || (value.IsUint() && matchesUint)
177  || (value.IsInt() && matchesInt)
178  || (value.IsUint64() && matchesUint64)
179  || (value.IsInt64() && matchesInt64)
180  || (value.IsNumber() && matchesNumber)
181  || (value.IsBool() && matchesBool));
182  }
183 
184 private:
185  const bool matchesObject;
186  const bool matchesArray;
187  const bool matchesNull;
188  const bool matchesInt;
189  const bool matchesInt64;
190  const bool matchesUint;
191  const bool matchesUint64;
192  const bool matchesNumber;
193  const bool matchesString;
194  const bool matchesBool;
195 };

242 Read(const rapidjson::Value& value, T& var)

251 // wstring
252 template <typename T>
253 typename boost::enable_if<is_wstring<T> >::type
254 Read(const rapidjson::Value& value, T& var)
255 {
256  std::basic_string<uint16_t> str =
257  boost::locale::conv::utf_to_utf<uint16_t>(
258  value.GetString(), value.GetString() + value.GetStringLength(), boost::locale::conv::stop);
259  const uint32_t length = static_cast<uint32_t>(str.size());
260 
261  resize_string(var, length);
262  std::copy(str.begin(), str.end(), string_data(var));
263 }
