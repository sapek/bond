23 // Visitor which applies protocol's parser to specified transform and data.
24 // It is used to dispatch to appropriate protocol at runtime.
25 template <typename T, typename Schema, typename Transform>
26 class _Parser
27  : public boost::static_visitor<bool>,
28  boost::noncopyable

30 public:
31  _Parser(const Transform& transform, const Schema& schema)
32  : _transform(transform),
33  _schema(schema)
34  {}

38  typename boost::enable_if<is_protocol_enabled<typename remove_const<Reader>::type>, bool>::type
39  operator()(Reader& reader) const

41  // Apply transform to serialized data
42  return Apply(_transform, reader, _schema, false);
43  }

45  template <typename Reader>
46  typename boost::disable_if<is_protocol_enabled<typename remove_const<Reader>::type>, bool>::type
47  operator()(Reader& /*reader*/) const

50  BOOST_ASSERT(false);

54  template <template <typename U> class Reader, typename Writer>
55  static
56  typename boost::enable_if_c<is_protocol_same<Reader<InputBuffer>, Writer>::value
57  && protocol_has_multiple_versions<Reader<InputBuffer> >::value, bool>::type
58  Apply(const Serializer<Writer>& transform, Reader<InputBuffer>& reader, const Schema& schema, bool base)
59  {
60  if (is_protocol_version_same(reader, transform._output))
61  return FastPassThrough(reader, transform._output, schema);
62  else
63  return typename Reader<InputBuffer>::Parser(reader, base).Apply(transform, schema);
64  }
65 
66  template <template <typename U> class Reader, typename Writer>
67  static
68  typename boost::enable_if_c<is_protocol_same<Reader<InputBuffer>, Writer>::value
69  && !protocol_has_multiple_versions<Reader<InputBuffer> >::value, bool>::type
70  Apply(const Serializer<Writer>& transform, Reader<InputBuffer>& reader, const Schema& schema, bool base)
71  {
72  BOOST_VERIFY(!base);
73  // Triggering the following assert means that bond::enable_protocol_versions trait is
74  // defined/specialized inconsistently for the protocol in different compilation units.
75  BOOST_ASSERT(is_protocol_version_same(reader, transform._output));
76  return FastPassThrough(reader, transform._output, schema);
77  }
78 
79  template <typename TransformT, typename Reader>
80  static
81  bool Apply(const TransformT& transform, Reader& reader, const Schema& schema, bool base)
82  {
83  return typename Reader::Parser(reader, base).Apply(transform, schema);
84  }
85 
86 protected:
87  template <typename Reader, typename Writer, typename SchemaT>
88  static
89  bool FastPassThrough(Reader& reader, Writer& writer, const SchemaT&)
90  {
91  bonded<T, Reader&> value(reader);
92  PassThrough(value, reader, writer);
93  return false;
94  }
95 
96  template <typename Reader, typename Writer>
97  static
98  bool FastPassThrough(Reader& reader, Writer& writer, const RuntimeSchema& schema)
99  {
100  bonded<void, Reader&> value(reader, schema);
101  PassThrough(value, reader, writer);
102  return false;
103  }

105  const Transform& _transform;

196 template <typename T, typename Transform, typename Buffer, typename Schema>
197 inline bool Parse(const Transform& transform, ProtocolReader<Buffer> reader, const Schema& schema, const RuntimeSchema* runtime_schema, bool base)
198 {
199  BOOST_VERIFY(!base);
200 
201  if (runtime_schema)
202  {
203  // Use named variable to avoid gcc silently copying objects (which
204  // causes build break, because Parser<> is non-copyable).
205  Parser<void, RuntimeSchema, Transform> parser(transform, *runtime_schema);
206  return boost::apply_visitor(parser, reader.value);
207  }
208  else
209  {
210  // Use named variable to avoid gcc silently copying objects (which
211  // causes build break, because Parser<> is non-copyable).
212  Parser<T, Schema, Transform> parser(transform, schema);
213  return boost::apply_visitor(parser, reader.value);
214  }

233  operator()(Reader<Buffer>& reader) const
234  {
